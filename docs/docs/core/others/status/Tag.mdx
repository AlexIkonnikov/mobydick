Reusable Status component

## Example

```tsx
import {
  IStatusState,
  IStatusType,
  Status,
} from '@lad-tech/mobydick-core';


export const StatusWidget = () => {

  return (
      <Status type={IStatusType.tag} state={IStatusState.green} text={'tag'} />
  );
};

```
