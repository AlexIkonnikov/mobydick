Works with ChatInput component


## Example

```tsx
import {useState} from 'react';

import {
  ChatInputField
} from '@lad-tech/mobydick-core';

export const ChatWidget = () => {
  const [input, setInput] = useState('');

  return (
    <ChatInputField value={input} onChangeText={setInput} />
  );
};

```
