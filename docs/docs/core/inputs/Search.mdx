## Example

```tsx
import {useState} from 'react';

import {Search, SimpleIcon} from '@lad-tech/mobydick-core';

const ExampleSearch = () => {
  const [value, setValue] = useState('');

  return (
      <Search
        value={value}
        onChangeText={setValue}
        leftIcon={<SimpleIcon name={'icon-search'} />}
      />
  );
};



export default ExampleSearch;

```
