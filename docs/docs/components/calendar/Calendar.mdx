Reusable Calendar

## Example

```tsx
import {Calendar, createStyles, useStyles, View} from '@lad-tech/mobydick-calendar';

export const CalendarWidget = () => {
  const [styles] = useStyles(stylesFn);

  return (
    <View style={styles.container}>
      <Calendar />
    </View>
  );
};

const stylesFn = createStyles(({spaces}) => ({
  container: {
    gap: spaces.Space16,
    alignItems: 'center',
  },
}));
```
