Reusable ModalCalendar

## Example

```tsx
import {Button, createStyles, usePopups, useStyles, View} from 'shared/ui';
import {ModalCalendar} from '@lad-tech/mobydick-calendar';

export const ModalCalendarWidget = () => {
  const [styles] = useStyles(stylesFn);
  const {openPopup} = usePopups();

  return (
    <View style={styles.container}>
      <Button
        text={'ModalCalendar'}
        onPress={() =>
          openPopup({
            Content: props => <ModalCalendar {...props} />,
          })
        }
      />
    </View>
  );
};

const stylesFn = createStyles(({spaces}) => ({
  container: {
    gap: spaces.Space16,
    alignItems: 'center',
  },
}));
```
